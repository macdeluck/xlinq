66  JoinLookup(std::shared_ptr<IEnumerator<std::shared_ptr<IGrouping<TInnerKey, TInnerElem>>>> innerGroupEnumerator,
67  TOuter outer, TOuterKeySelector outerKeySelector, TResultSelector resultSelector, TKeyEqComp keyEqComp, TOuterHasher outerHasher, TOuterEqComp outerEqComp)
68  : _outer(outer), _innerGroupEnumerator(innerGroupEnumerator),
69  _outerGroupEnumerator(from(_outer) >> group_by(outerKeySelector, outerHasher, outerEqComp) >> getEnumerator()), _outerSkipped(std::unordered_map<TOuterKey, std::shared_ptr<IEnumerable<TOuterElem>>, TOuterHasher, TOuterEqComp>()), _resultSelector(resultSelector), _result(std::list<TResult>()),
70  _keyEqComp(keyEqComp)
71  {
72  }
73 
74  bool finished()
75  {
76  return !_innerGroupEnumerator && !_outerGroupEnumerator;
77  }
78 
79  bool next()
80  {
81  if (finished()) return false;
82  while (_innerGroupEnumerator && _innerGroupEnumerator->next())
83  {
84  auto current = _innerGroupEnumerator->current();
85  auto outerCurrent = _outerSkipped.find(current->getKey());
86  if (outerCurrent != _outerSkipped.end())
87  {
88  for (auto inIt = current->getEnumerator(); inIt->next();)
89  {
90  for (auto it = (*outerCurrent).second->getEnumerator(); it->next();)
91  {
92  _result.push_back(_resultSelector(inIt->current(), it->current()));
93  }
94  }
95  _outerSkipped.erase(current->getKey());
96  return true;
97  }
98  else
99  {
100  while (_outerGroupEnumerator && _outerGroupEnumerator->next())
101  {
102  if (_keyEqComp(current->getKey(), _outerGroupEnumerator->current()->getKey()))
103  {
104  for (auto inIt = current->getEnumerator(); inIt->next();)
105  {
106  for (auto it = _outerGroupEnumerator->current()->getEnumerator(); it->next();)
107  {
108  _result.push_back(_resultSelector(inIt->current(), it->current()));
109  }
110  }
111  return true;
112  }
113  else
114  {
115  _outerSkipped.insert({ _outerGroupEnumerator->current()->getKey(), _outerGroupEnumerator->current() });
116  }
117  }
118  _outerGroupEnumerator = nullptr;
119  }
120  }
121  _innerGroupEnumerator = nullptr;
122  _outerSkipped.clear();
123  return false;
124  }
125 
126  typename std::list<TResult>::const_iterator begin() { return _result.begin(); }
127 
128  typename std::list<TResult>::const_iterator end() { return _result.end(); }
